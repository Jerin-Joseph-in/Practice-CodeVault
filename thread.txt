class apple
{
 public  void orange(int n)
    {
        synchronized(this)
        {
        for(int i=1;i<n;i++)
        {
            System.out.println(n*i);
  try  {
        Thread.sleep(100);
}
catch(Exception e)
{
    System.out.println(e);
}
}}
}}
class rose extends Thread
{
    apple a;
   public rose(apple a)
    {
        this.a=a;
    }
  public  void run()
    {
        a.orange(5);
    }
}
