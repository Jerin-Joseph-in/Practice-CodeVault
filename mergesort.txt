#include <stdio.h>

void merge(int a[],int low,int mid,int high);
void mergesort(int a[],int low,int high);

void mergesort(int a[],int low,int high)
{  int mid;
    if(low<high)
    {
        mid=(low+high)/2;
        mergesort(a,low,mid);
        mergesort(a,mid+1,high);
        merge(a,low,mid,high);
    }
}
void merge(int a[],int low,int mid,int high)
{ int i,j,k,b[30];
    i=low;
    j=mid+1;
    k=0;
    while(i<=mid && j<=high)
    {
        if(a[i]<=a[j])
        {
            b[k]=a[i];
            k++;
            i++;
        }
        else
        {
            b[k]=a[j];
            j++;
            k++;
        }
    }
    while(i<=mid)
    {
        b[k]=a[i];
            k++;
            i++;
    }
    while(j<=high)
    {b[k]=a[j];
            j++;
            k++;
        
    }
    for(i=0;i<k;i++)
    {
        a[low+i]=b[i];
    }
}
